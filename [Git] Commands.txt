# initialization
- git init

# clone
- git clone {주소} {폴더명}

# create branch
- git checkout -b {브랜치명}

# pull
- git pull origin {브랜치명}

## Process of Uploading codes ##
# add
- git add {파일명}
- git add .

# (opt) check status
- git status

# commit & make history
- git commit -m "{history}"

# connect local project with repository
- git remote add origin {주소}

# (opt) check connection
- git remote -v

# push
- git push origin {브랜치명}

# Simple Process
- add > commit > push
